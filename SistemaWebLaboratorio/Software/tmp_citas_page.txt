"use client";
import { useEffect, useMemo, useState } from 'react';
import { Header } from '../components/header';
import { Button } from '../../../components/ui/button';
import { StatusBadge } from '../../../components/ui/status-badge';
import { CalendarIcon } from '../../../components/ui/icons';
import { useSessionStore } from '../../../lib/session-store';
import { api } from '../../../lib/api';

type Slot = { slot_id:number; inicio:string; fin:string };
type Cita = { numero_cita:number; codigo_servicio:string; estado:string; inicio:string; fin:string };

export default function Page(){
  const { accessToken } = useSessionStore();
  const [svc] = useState('LAB_GENERAL');
  const today = useMemo(()=> new Date(), []);
  const [desde, setDesde] = useState<string>(()=> new Date(today.getFullYear(), today.getMonth(), today.getDate()).toISOString());
  const [hasta, setHasta] = useState<string>(()=> new Date(today.getFullYear(), today.getMonth(), today.getDate()+7).toISOString());
  const [slots, setSlots] = useState<Slot[]>([]);
  const [citas, setCitas] = useState<Cita[]>([]);
  const [status, setStatus] = useState('');

  const load = async () => {
    if(!accessToken) return;
    const qs = `?codigo_servicio=${encodeURIComponent(svc)}&desde=${encodeURIComponent(desde)}&hasta=${encodeURIComponent(hasta)}`;
    const res = await api<{ items: Slot[] }>(`/appointments/disponibilidad${qs}`, { method:'GET' }, accessToken);
    setSlots(res.items || []);
    const mine = await api<{ items: Cita[] }>(`/appointments/mis-citas`, { method:'GET' }, accessToken);
    setCitas(mine.items || []);
  };

  useEffect(()=>{ load(); }, [accessToken, desde, hasta]);

  const reservar = async (slot_id:number) => {
    if(!accessToken) return;
    try{
      await api(`/appointments/crear`, { method:'POST', body: JSON.stringify({ codigo_servicio: svc, slot_id }) }, accessToken);
      setStatus('Cita creada');
      await load();
    }catch(e:any){ setStatus(String(e?.message || 'Error')); }
  };

  const reprogramar = async (numero_cita:number, nuevo_slot_id:number) => {
    if(!accessToken) return;
    try{
      await api(`/appointments/reprogramar`, { method:'POST', body: JSON.stringify({ numero_cita, nuevo_slot_id }) }, accessToken);
      setStatus('Cita reprogramada');
      await load();
    }catch(e:any){ setStatus(String(e?.message || 'Error')); }
  };

  return (
    <section aria-label="citas">
      <Header title="Gestión de Citas" subtitle="Agenda y gestiona tus citas médicas" />
      <div className="row cols-2 mt-4">
        <div className="card">
          <div className="heading-sm flex items-center gap-2"><CalendarIcon /> Disponibilidad</div>
          <div className="grid grid-cols-1 md:grid-cols-3 gap-2 mt-2">
            <label className="space-y-1"><span>Desde</span><input type="date" value={new Date(desde).toISOString().slice(0,10)} min={new Date().toISOString().slice(0,10)} onChange={(e)=>setDesde(new Date(e.target.value).toISOString())} /></label>
            <label className="space-y-1"><span>Hasta</span><input type="date" value={new Date(hasta).toISOString().slice(0,10)} min={new Date().toISOString().slice(0,10)} onChange={(e)=>setHasta(new Date(e.target.value).toISOString())} /></label>
            <div className="flex items-end"><Button onClick={load}>Actualizar</Button></div>
          </div>
          <div className="mt-3 space-y-2">
            {slots.map(s => (
              <div key={s.slot_id} className="flex items-center justify-between rounded-md border border-[var(--border-soft)] bg-white p-3">
                <div>
                  <div className="font-medium">{new Date(s.inicio).toLocaleString()} â€” {new Date(s.fin).toLocaleTimeString()}</div>
                  <div className="body-muted">{svc}</div>
                </div>
                <Button onClick={()=>reservar(s.slot_id)}>Reservar</Button>
              </div>
            ))}
            {!slots.length && <div className="subtitle">Sin disponibilidad para el rango.</div>}
          </div>
        </div>
        <div className="card">
          <div className="heading-sm">Mis Citas</div>
          <div className="body-muted">Lista de citas agendadas</div>
          <div className="mt-3 space-y-2">
            {citas.map(c => (
              <div key={c.numero_cita} className="flex items-center justify-between rounded-md border border-[var(--border-soft)] bg-white p-3">
                <div>
                  <div className="font-medium">{c.codigo_servicio} â€” #{c.numero_cita}</div>
                  <div className="body-muted">{new Date(c.inicio).toLocaleString()} â€” {new Date(c.fin).toLocaleTimeString()}</div>
                </div>
                <div className="flex items-center gap-2">
                  <StatusBadge status={c.estado as any} />
                  {slots[0] && <Button variant="outline" onClick={()=>reprogramar(c.numero_cita, slots[0].slot_id)}>Reprogramar</Button>}
                </div>
              </div>
            ))}
            {!citas.length && <div className="subtitle">Aún no tienes citas.</div>}
          </div>
        </div>
      </div>
      {status && <div className="card mt-3 subtitle">{status}</div>}
    </section>
  );
}
